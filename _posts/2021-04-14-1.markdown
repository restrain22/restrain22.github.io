'''c
#include<iostream>
#include<vector>
#include<algorithm>
#pragma warning(disable:4996)
using namespace std;

int main() {
	int N = 0; int xMAX, yMAX, xMIN, yMIN, dir;
	
	dir = 2; //0:북 1:동 2:남 3:서 방향
	scanf("%d", &N);
	vector<vector<char>> graph(2 * N+1, vector<char>(2 * N+1, '#'));
	int x, y;
	x = y = N;
	//벡터에 값 저장
	graph[x][y] = '.';
	xMAX = yMAX = xMIN = yMIN = N;
	for (int i = 0; i < N; i++)
	{
		//printf("xMIN: %d / xMAX: %d\n yMIN: %d / yMAX: %d\n",xMIN,xMAX,yMIN,yMAX);

		char ch = ' ';
		scanf(" %c", &ch);
		if (ch == 'R')
			dir = (dir + 1) % 4;
		else if (ch == 'L')
			dir = (4+dir - 1) % 4;
		else {
			switch (dir)
			{
			case 0:
				y = y - 1;
				graph[y][x] = '.';
				yMIN = min(y, yMIN);
				break;
			case 1:
				x = x + 1;
				graph[y][x] = '.';
				xMAX = max(x, xMAX);
				break;
			case 2:
				y = y + 1;
				graph[y][x] = '.';
				yMAX = max(y, yMAX);
				break;
			case 3:
				x = x - 1;
				graph[y][x] = '.';
				xMIN = min(x, xMIN);
				break;
			}
		}
	}
	for (int i = yMIN; i <= yMAX; i++)
	{
		for (int j = xMIN; j <= xMAX; j++)
		{
			printf("%c", graph[i][j]);
		}printf("\n");

	}
}
'''
